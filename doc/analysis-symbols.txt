%%%%%%% (mostly) Nominal morphemes

%
% Nominal plural suffix 
%
<plu>   - Plural suffix            -- ev-ler    `houses'
        G&K:        -PL
        Oflazer:    +PL

%
% Nominal possessive suffixes
%

<p1s>   - 1st person singular possessive    -- ev-im `my house'
        G&K:        -1S.POSS
        Oflazer:    +POS1S

<p2s>   - 2nd person singular possessive    -- ev-in `your house'
        G&K:        -2S.POSS
        Oflazer:    +POS2S

<p3s>   - 3rd person singular possessive    -- ev-i `his/her/its house'
        G&K:        -3S.POSS
        Oflazer:    +POS3S

<p1p>   - 1st person plural possessive  -- ev-imiz `our house'
        G&K:        -1P.POSS
        Oflazer:    +POS1P

<p2p>   - 2st person plural possessive  -- ev-iniz `your(plural/formal) house'
        G&K:        -2P.POSS
        Oflazer:    +POS2P

<p3p>   - 3st person plural possessive  -- ev-leri `their house'
        G&K:        -3P.POSS
        Oflazer:    +POS3P

%
% Case suffixes (or postpositions)
%
         
<loc>   - Locative case     -- ev-de `in/on/at the house'
        G&K:        -LOC
        Oflazer:    +Loc

<gen>   - Genitive case     -- ev-in `the one that belongs the house'
        G&K:        -GEN
        Oflazer:    +Gen

<acc>   - Accusative case   -- ev-i 
        G&K:        -ACC
        Oflazer:    +Acc

<abl>   - Ablative case     -- ev-den   `from the house'
        G&K:        -ABL
        Oflazer:    +Abl

<dat>   - Dative case       --- ev-e    `to the house'
        G&K:        -DAT
        Oflazer:    +Dat

<ins>   - Instrumental/comitative -- ev-le  `with the house'
        G&K:        -INS
        Oflazer:    +Ins


%
% -ki attaches to the nominals in locative and genitive case and 
% forms "attributive adjectival phrases or pronominal expressions" (G&K).
% It can again get one of these case suffixes and recurse. 
%

<ki>    -  -- ev-de-ki-nin-ki  `the one that belons to the person in the house' 
        G&K:        -ADJ/-PRON
        Oflazer:    +Rel

%%%%% morphemes that attach to (mostly) verbs

%
% voice suffixes
%
<refl>  - Reflexive   -- yıka `to wash' -> yıka-n `to wash oneself'
        G&K:        -REF
        Oflazer:    +Refl

<recp>  - Reciprocal  -- sev `to love' -> sev-iş `to love each other/make love'
        G&K:        -REC
        Oflazer:    +Recip

<caus> - Causative, can be multiple -- yika-t-tır `to make someone have (something) washed'
        G&K:        -CAUS
        Oflazer:    +Caus

<pass> - Passive -- sev-il `to be loved'
        G&K:        -PASS
        Oflazer:    +Pass
